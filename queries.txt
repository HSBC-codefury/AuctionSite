CREATE TABLE AUCTION_User(
User_Id INT NOT NULL,
Name VARCHAR(100) NOT NULL,
DOB DATE NOT NULL,
Email VARCHAR(100) NOT NULL,
Contact_No BIGINT NOT NULL,
User_Name VARCHAR(100) NOT NULL, 
Password VARCHAR(100) NOT NULL,
Address VARCHAR(100) NOT NULL, 
User_Type VARCHAR(100) NOT NULL,
Wallet_Amount DOUBLE, 
CONSTRAINT User_Id_PK PRIMARY KEY(User_Id));



CREATE TABLE CATEGORY(
Unique_Id INT NOT NULL, 
Category_Name VARCHAR(100) NOT NULL,
Category_Description VARCHAR(500) NOT NULL, 
CONSTRAINT Category_Name_CPK PRIMARY KEY(Category_Name));

CREATE TABLE PRODUCT
(Unique_Id INT NOT NULL, 
Product_Name VARCHAR(100) NOT NULL,
Product_Category VARCHAR(100) NOT NULL , 
Product_Description VARCHAR(500) NOT NULL, 
Actual_Price DOUBLE NOT NULL, 
Quantity INT NOT NULL, 
User_Id INT NOT NULL, 
CONSTRAINT Unique_Id_PPK PRIMARY KEY(Unique_Id),
CONSTRAINT Product_Category_CFK FOREIGN KEY(Product_Category) REFERENCES CATEGORY(Category_Name),
CONSTRAINT User_Id_FK FOREIGN KEY(User_Id) REFERENCES AUCTION_USER(User_Id));

CREATE TABLE SCHEDULE_AUCTION(
Unique_Id INT NOT NULL, 
Product_Id INT NOT NULL,
Min_Bid_Value DOUBLE NOT NULL, 
Bid_Start_Date DATE NOT NULL, 
Bid_End_Date DATE NOT NULL, 
User_Id INT NOT NULL,
Sold_Price DOUBLE NOT NULL, 
Bid_Status VARCHAR(100) NOT NULL, 
CONSTRAINT Product_Id_APFK FOREIGN KEY(Product_Id) REFERENCES PRODUCT(Unique_Id), 
CONSTRAINT User_Id_AFK FOREIGN KEY(User_Id) REFERENCES AUCTION_USER(User_Id));

CREATE TABLE BID(
Unique_Id INT NOT NULL, 
Product_Id INT NOT NULL, 
Bid_Value DOUBLE NOT NULL, 
User_Id INT NOT NULL, 
Bid_Status VARCHAR(100) NOT NULL, 
CONSTRAINT Unique_Id_PKK PRIMARY KEY(Unique_Id),
CONSTRAINT Product_Id_BPFK FOREIGN KEY(Product_Id) REFERENCES PRODUCT(Unique_Id), 
CONSTRAINT User_Id_BFK FOREIGN KEY(User_Id) REFERENCES AUCTION_USER(User_Id));
